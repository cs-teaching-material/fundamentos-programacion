Helper

Decision simple reusable

static bool EsCantidadMayorista(int cantidad) { return cantidad >= 10; }

Por que helper

  • Evita repetir condicion en varios lugares.
  • Hace el codigo mas legible.
  • Permite cambiar regla sin tocar todo.

Codigo completo final

Programa funcional end-to-end

using System; class Program { static bool EsCantidadMayorista(int cantidad) { return cantidad >= 10; } static double CalcularSubtotal(int cantidad, double precioUnitario) { return cantidad * precioUnitario; } static double CalcularDescuento(int cantidad, double subtotal) { if (EsCantidadMayorista(cantidad)) { return subtotal * 0.10; } return 0; } static double CalcularIVA(double baseImponible) { return baseImponible * 0.19; } static double CalcularTotal(double baseImponible, double iva) { return baseImponible + iva; } static bool LeerEnteroPositivo(string mensaje, out int valor) { Console.Write(mensaje); if (!int.TryParse(Console.ReadLine(), out valor)) { return false; } return valor > 0; } static bool LeerDoublePositivo(string mensaje, out double valor) { Console.Write(mensaje); if (!double.TryParse(Console.ReadLine(), out valor)) { return false; } return valor > 0; } static void MostrarResumen(string producto, int cantidad, double precio, double subtotal, double descuento, double iva, double total) { Console.WriteLine("--- Resumen de venta ---"); Console.WriteLine("Producto: " + producto); Console.WriteLine("Cantidad: " + cantidad); Console.WriteLine("Precio unitario: " + precio); Console.WriteLine("Subtotal: " + subtotal); Console.WriteLine("Descuento: " + descuento); Console.WriteLine("IVA: " + iva); Console.WriteLine("Total: " + total); } static void RegistrarPedido() { Console.Write("Producto: "); string producto = Console.ReadLine(); if (!LeerEnteroPositivo("Cantidad: ", out int cantidad) || !LeerDoublePositivo("Precio unitario: ", out double precioUnitario)) { Console.WriteLine("Entrada invalida. Verifica los datos."); return; } double subtotal = CalcularSubtotal(cantidad, precioUnitario); double descuento = CalcularDescuento(cantidad, subtotal); double baseImponible = subtotal - descuento; double iva = CalcularIVA(baseImponible); double total = CalcularTotal(baseImponible, iva); MostrarResumen(producto, cantidad, precioUnitario, subtotal, descuento, iva, total); } static void Main() { RegistrarPedido(); } }
